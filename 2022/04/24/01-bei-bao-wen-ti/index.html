1049. 最后一块石头的重量 II

乍看毫无思路，但仔细想想，这就是相当于把石头分成两堆，两堆相互碰撞，最后剩下的重量最小。那么就可以转化为01背包问题。

class Solution {
public:
    int lastStoneWeightII(vector<int>& stones) {
        int n = stones.size();
        int sum = accumulate(stones.begin(), stones.end(), 0);
        int tgt = sum / 2;
        vector<int> dp(tgt + 1);    // dp[i]表示容量（可容纳的重量）为i的背包最多可以装下重量为dp[i]的石头
        for (int i = 0; i < n; i++) {   // 正序遍历物品
            for (int j = tgt; j >= stones[i]; j--) {    // 倒序遍历背包
                dp[j] = max(dp[j], dp[j - stones[i]] + stones[i]);
            }
        }
        return sum - dp[tgt] - dp[tgt];
    }
};

class Solution {
public:
    int findTargetSumWays(vector<int>& nums, int target) {
        int n = nums.size();
        int sum = accumulate(nums.begin(), nums.end(), 0);
        if (abs(target) > sum || (target + sum) % 2 == 1) {
            return 0;
        }
        int positive = (target + sum) / 2;  // (target + sum) 有可能小于零
        vector<int> dp(positive + 1);
        dp[0] = 1;
        for (int i = 0; i < n; i++) {
            for (int j = positive; j >= nums[i]; j--) {
                dp[j] += dp[j - nums[i]];
            }
        }
        return dp[positive];
    }
};
/*
    x - (sum - x) = target
->  2x = target + sum
->  x = (target + sum) / 2
正数为x, 负数为-(sum - x)
要满足 x - (sum - x) = target
需  (target + sum) % 2 == 0

dp[i]表示和为i的方案数

474. 一和零（二维0-1背包）

前面的题目中背包容量都是一维的，要么是价值，要么是重量。在这个题目中，背包容量变成了二维，也就是 [0的个数, 1的个数]。

同样的方法，类比一维背包，第一层循环遍历物品，第二层循环遍历背包容量，这里只不过把第二层循环扩展到二维，变成：第一层循环遍历数字，第二层遍历背包的第一个维度，第三层遍历背包的第二个维度（这两个维度没有顺序限制）。与一维背包相同，遍历的容量的时候也要倒着遍历。代码如下：

class Solution {
public:
    int findMaxForm(vector<string>& strs, int m, int n) {
        vector<vector<int>> dp(m + 1, vector<int>(n + 1));
        for (auto s : strs) {
            int cnt0 = 0, cnt1 = 0;
            for (auto ch : s) {
                if (ch == '0') {
                    cnt0++;
                } else {
                    cnt1++;
                }
            }
            for (int i = m; i >= cnt0; i--) {
                for (int j = n; j >= cnt1; j--) {
                    dp[i][j] = max(dp[i][j], dp[i - cnt0][j - cnt1] + 1);
                }
            }
        }
        return dp[m][n];
    }
};
